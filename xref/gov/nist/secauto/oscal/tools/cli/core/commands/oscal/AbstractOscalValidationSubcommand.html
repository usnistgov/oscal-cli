27  package gov.nist.secauto.oscal.tools.cli.core.commands.oscal;
28  
29  import gov.nist.secauto.metaschema.binding.IBindingContext;
30  import gov.nist.secauto.metaschema.cli.commands.AbstractValidateContentCommand;
31  import gov.nist.secauto.metaschema.cli.processor.CLIProcessor.CallingContext;
32  import gov.nist.secauto.metaschema.cli.processor.command.ICommandExecutor;
33  import gov.nist.secauto.metaschema.model.common.constraint.IConstraintSet;
34  import gov.nist.secauto.oscal.lib.OscalBindingContext;
35  
36  import org.apache.commons.cli.CommandLine;
37  import org.json.JSONObject;
38  
39  import java.io.IOException;
40  import java.util.List;
41  import java.util.Set;
42  
43  import javax.xml.transform.Source;
44  
45  import edu.umd.cs.findbugs.annotations.NonNull;
46  
47  public abstract class AbstractOscalValidationSubcommand
48      extends AbstractValidateContentCommand {
49  
50    @NonNull
51    protected abstract List<Source> getOscalXmlSchemas() throws IOException;
52  
53    @NonNull
54    protected abstract JSONObject getOscalJsonSchema();
55  
56    @Override
57    public ICommandExecutor newExecutor(CallingContext callingContext, CommandLine commandLine) {
58      return new OscalCommandExecutor(callingContext, commandLine);
59    }
60  
61    private class OscalCommandExecutor
62        extends AbstractValidationCommandExecutor {
63  
64      private OscalCommandExecutor(
65          @NonNull CallingContext callingContext,
66          @NonNull CommandLine commandLine) {
67        super(callingContext, commandLine);
68      }
69  
70      @Override
71      protected IBindingContext getBindingContext(@NonNull Set<IConstraintSet> constraintSets) {
72        return constraintSets.isEmpty() ? OscalBindingContext.instance() : new OscalBindingContext(constraintSets);
73      }
74  
75      @Override
76      @NonNull
77      public List<Source> getXmlSchemas() throws IOException {
78        return getOscalXmlSchemas();
79      }
80  
81      @Override
82      @NonNull
83      public JSONObject getJsonSchema() {
84        return getOscalJsonSchema();
85      }
86    }
87  }
