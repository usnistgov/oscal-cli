128   @SuppressWarnings({
129       "PMD.CyclomaticComplexity", "PMD.CognitiveComplexity", // reasonable
130       "PMD.PreserveStackTrace" // intended

132   @Override
133   public void validateOptions(CallingContext callingContext, CommandLine cmdLine) throws InvalidArgumentException {
134     if (cmdLine.hasOption(AS_OPTION)) {
135       try {
136         String toFormatText = cmdLine.getOptionValue(AS_OPTION);
137         Format.valueOf(toFormatText.toUpperCase(Locale.ROOT));
138       } catch (IllegalArgumentException ex) {
139         InvalidArgumentException newEx = new InvalidArgumentException(
140             String.format("Invalid '%s' argument. The format must be one of: %s.",
141                 OptionUtils.toArgument(AS_OPTION),
142                 Arrays.asList(Format.values()).stream()
143                     .map(format -> format.name())
144                     .collect(CustomCollectors.joiningWithOxfordComma("and"))));
145         newEx.setOption(AS_OPTION);
146         newEx.addSuppressed(ex);
147         throw newEx;
148       }
149     }
150 
151     if (cmdLine.hasOption(TO_OPTION)) {
152       try {
153         String toFormatText = cmdLine.getOptionValue(TO_OPTION);
154         Format.valueOf(toFormatText.toUpperCase(Locale.ROOT));
155       } catch (IllegalArgumentException ex) {
156         InvalidArgumentException newEx
157             = new InvalidArgumentException("Invalid '--to' argument. The format must be one of: "
158                 + Arrays.asList(Format.values()).stream()
159                     .map(format -> format.name())
160                     .collect(CustomCollectors.joiningWithOxfordComma("and")));
161         newEx.setOption(AS_OPTION);
162         newEx.addSuppressed(ex);
163         throw newEx;
164       }
165     }
166 
167     List<String> extraArgs = cmdLine.getArgList();
168     if (extraArgs.isEmpty()) {
169       throw new InvalidArgumentException("The source to resolve must be provided.");
170     }
171 
172     File source = new File(extraArgs.get(0));
173     if (!source.exists()) {
174       throw new InvalidArgumentException("The provided source '" + source.getPath() + "' does not exist.");
175     }
176     if (!source.canRead()) {
177       throw new InvalidArgumentException("The provided source '" + source.getPath() + "' is not readable.");
178     }
179   }

183     return ICommandExecutor.using(callingContext, cmdLine, this::executeCommand);

186   @SuppressWarnings({
187       "PMD.OnlyOneReturn", // readability
188       "unused"
189   })
190   protected ExitStatus executeCommand(
191       @NonNull CallingContext callingContext,
192       @NonNull CommandLine cmdLine) {
193     List<String> extraArgs = cmdLine.getArgList();
194     Path source = resolvePathAgainstCWD(ObjectUtils.notNull(Paths.get(extraArgs.get(0))));
195 
196     IBoundLoader loader = OscalBindingContext.instance().newBoundLoader();
197     loader.disableFeature(DeserializationFeature.DESERIALIZE_VALIDATE_CONSTRAINTS);
198 
199     Format asFormat;
200     // attempt to determine the format
201     if (cmdLine.hasOption(AS_OPTION)) {
202       try {
203         String asFormatText = cmdLine.getOptionValue(AS_OPTION);
204         asFormat = Format.valueOf(asFormatText.toUpperCase(Locale.ROOT));
205       } catch (IllegalArgumentException ex) {
206         return ExitCode.INVALID_ARGUMENTS
207             .exitMessage("Invalid '--as' argument. The format must be one of: " + Arrays.stream(Format.values())
208                 .map(format -> format.name())
209                 .collect(CustomCollectors.joiningWithOxfordComma("or")));
210       }
211     } else {
212       // attempt to determine the format
213       try {
214         asFormat = loader.detectFormat(ObjectUtils.notNull(source));
215       } catch (FileNotFoundException ex) {
216         // this case was already checked for
217         return ExitCode.IO_ERROR.exitMessage("The provided source file '" + source + "' does not exist.");
218       } catch (IOException ex) {
219         return ExitCode.PROCESSING_ERROR.exit().withThrowable(ex);
220       } catch (IllegalArgumentException ex) {
221         return ExitCode.INVALID_ARGUMENTS.exitMessage(
222             "Source file has unrecognizable format. Use '--as' to specify the format. The format must be one of: "
223                 + Arrays.stream(Format.values())
224                     .map(format -> format.name())
225                     .collect(CustomCollectors.joiningWithOxfordComma("or")));
226       }
227     }
228 
229     source = source.toAbsolutePath();
230 
231     Format toFormat;
232     if (cmdLine.hasOption(TO_OPTION)) {
233       String toFormatText = cmdLine.getOptionValue(TO_OPTION);
234       toFormat = Format.valueOf(toFormatText.toUpperCase(Locale.ROOT));
235     } else {
236       toFormat = asFormat;
237     }
238 
239     Path destination = null;
240     if (extraArgs.size() == 2) {
241       destination = Paths.get(extraArgs.get(1)).toAbsolutePath();
242     }
243 
244     if (destination != null) {
245       if (Files.exists(destination)) {
246         if (!cmdLine.hasOption(OVERWRITE_OPTION)) {
247           return ExitCode.INVALID_ARGUMENTS.exitMessage("The provided destination '" + destination
248               + "' already exists and the --overwrite option was not provided.");
249         }
250         if (!Files.isWritable(destination)) {
251           return ExitCode.IO_ERROR.exitMessage("The provided destination '" + destination + "' is not writable.");
252         }
253       } else {
254         Path parent = destination.getParent();
255         if (parent != null) {
256           try {
257             Files.createDirectories(parent);
258           } catch (IOException ex) {
259             return ExitCode.INVALID_TARGET.exit().withThrowable(ex);
260           }
261         }
262       }
263     }
264 
265     IDocumentNodeItem document;
266     try {
267       document = loader.loadAsNodeItem(asFormat, loader.toInputSource(ObjectUtils.notNull(source.toUri())));
268     } catch (IOException ex) {
269       return ExitCode.IO_ERROR.exit().withThrowable(ex);
270     }
271     Object object = document.getValue();
272     if (object instanceof Catalog) {
273       // this is a catalog
274       return ExitCode.INVALID_ARGUMENTS.exitMessage("The target file is already a catalog");
275     } else if (object instanceof Profile) {
276       // this is a profile
277       StaticContext staticContext = new StaticContext();
278       URI sourceUri = ObjectUtils.notNull(source.toUri());
279       staticContext.setBaseUri(sourceUri);
280       DynamicContext dynamicContext = staticContext.newDynamicContext();
281       dynamicContext.setDocumentLoader(loader);
282       ProfileResolver resolver = new ProfileResolver();
283       resolver.setDynamicContext(dynamicContext);
284 
285       IDocumentNodeItem resolvedProfile;
286       try {
287         resolvedProfile = resolver.resolve(document);
288       } catch (IOException | ProfileResolutionException ex) {
289         return ExitCode.PROCESSING_ERROR
290             .exitMessage(
291                 String.format("Unable to resolve profile '%s'. %s", document.getDocumentUri(), ex.getMessage()))
292             .withThrowable(ex);
293       }
294 
295       // DefaultConstraintValidator validator = new DefaultConstraintValidator(dynamicContext);
296       // ((IBoundXdmNodeItem)resolvedProfile).validate(validator);
297       // validator.finalizeValidation();
298 
299       ISerializer<Catalog> serializer
300           = OscalBindingContext.instance().newSerializer(toFormat, Catalog.class);
301       try {
302         if (destination == null) {
303           serializer.serialize((Catalog) resolvedProfile.getValue(), ObjectUtils.notNull(System.out));
304         } else {
305           serializer.serialize((Catalog) resolvedProfile.getValue(), destination);
306         }
307       } catch (IOException ex) {
308         return ExitCode.PROCESSING_ERROR.exit().withThrowable(ex);
309       }
310     }
311     return ExitCode.OK.exit();
312   }
