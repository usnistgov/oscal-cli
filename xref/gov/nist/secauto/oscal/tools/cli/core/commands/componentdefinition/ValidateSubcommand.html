27  package gov.nist.secauto.oscal.tools.cli.core.commands.componentdefinition;
28  
29  import gov.nist.secauto.metaschema.binding.io.xml.XmlUtil;
30  import gov.nist.secauto.metaschema.model.common.util.CollectionUtil;
31  import gov.nist.secauto.metaschema.model.common.util.ObjectUtils;
32  import gov.nist.secauto.metaschema.model.common.validation.JsonSchemaContentValidator;
33  import gov.nist.secauto.oscal.lib.OscalBindingContext;
34  import gov.nist.secauto.oscal.tools.cli.core.commands.oscal.AbstractOscalValidationSubcommand;
35  
36  import org.json.JSONObject;
37  
38  import java.io.IOException;
39  import java.util.LinkedList;
40  import java.util.List;
41  
42  import javax.xml.transform.Source;
43  
44  public class ValidateSubcommand
45      extends AbstractOscalValidationSubcommand {
46    @Override
47    public String getDescription() {
48      return "Check that the specified OSCAL instance is well-formed and valid to the Component Definition model.";
49    }
50  
51    @Override
52    protected List<Source> getOscalXmlSchemas() throws IOException {
53      List<Source> retval = new LinkedList<>();
54      retval.add(
55          ObjectUtils.requireNonNull(
56              XmlUtil.getStreamSource(
57                  OscalBindingContext.class.getResource("/schema/xml/oscal-component-definition_schema.xsd"))));
58      return CollectionUtil.unmodifiableList(retval);
59    }
60  
61    @Override
62    protected JSONObject getOscalJsonSchema() {
63      return JsonSchemaContentValidator.toJsonObject(
64          ObjectUtils.requireNonNull(
65              OscalBindingContext.class.getResourceAsStream("/schema/json/oscal-component-definition_schema.json")));
66    }
67  }
