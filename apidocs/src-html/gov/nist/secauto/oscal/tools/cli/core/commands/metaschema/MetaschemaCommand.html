027package gov.nist.secauto.oscal.tools.cli.core.commands.metaschema;
028
029import com.google.auto.service.AutoService;
030
031import gov.nist.secauto.metaschema.cli.commands.GenerateSchemaCommand;
032import gov.nist.secauto.metaschema.cli.commands.ValidateContentWithMetaschemaCommand;
033import gov.nist.secauto.metaschema.cli.commands.ValidateMetaschemaCommand;
034import gov.nist.secauto.metaschema.cli.processor.command.AbstractParentCommand;
035import gov.nist.secauto.metaschema.cli.processor.command.ICommand;
036
037@AutoService(ICommand.class)
038public class MetaschemaCommand
039    extends AbstractParentCommand {
040  private static final String COMMAND = "metaschema";
041
042  public MetaschemaCommand() {
043    super(true);
044    addCommandHandler(new GenerateSchemaCommand());
045    addCommandHandler(new ValidateMetaschemaCommand());
046    addCommandHandler(new ValidateContentWithMetaschemaCommand());
047  }
048
049  @Override
050  public String getName() {
051    return COMMAND;
052  }
053
054  @Override
055  public String getDescription() {
056    return "Perform an operation on a Metaschema";
057  }
058}
