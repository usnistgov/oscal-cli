027package gov.nist.secauto.oscal.tools.cli.core.commands.oscal;
028
029import gov.nist.secauto.metaschema.binding.IBindingContext;
030import gov.nist.secauto.metaschema.cli.commands.AbstractValidateContentCommand;
031import gov.nist.secauto.metaschema.cli.processor.CLIProcessor.CallingContext;
032import gov.nist.secauto.metaschema.cli.processor.command.ICommandExecutor;
033import gov.nist.secauto.metaschema.model.common.constraint.IConstraintSet;
034import gov.nist.secauto.oscal.lib.OscalBindingContext;
035
036import org.apache.commons.cli.CommandLine;
037import org.json.JSONObject;
038
039import java.io.IOException;
040import java.util.List;
041import java.util.Set;
042
043import javax.xml.transform.Source;
044
045import edu.umd.cs.findbugs.annotations.NonNull;
046
047public abstract class AbstractOscalValidationSubcommand
048    extends AbstractValidateContentCommand {
049
050  @NonNull
051  protected abstract List<Source> getOscalXmlSchemas() throws IOException;
052
053  @NonNull
054  protected abstract JSONObject getOscalJsonSchema();
055
056  @Override
057  public ICommandExecutor newExecutor(CallingContext callingContext, CommandLine commandLine) {
058    return new OscalCommandExecutor(callingContext, commandLine);
059  }
060
061  private class OscalCommandExecutor
062      extends AbstractValidationCommandExecutor {
063
064    private OscalCommandExecutor(
065        @NonNull CallingContext callingContext,
066        @NonNull CommandLine commandLine) {
067      super(callingContext, commandLine);
068    }
069
070    @Override
071    protected IBindingContext getBindingContext(@NonNull Set<IConstraintSet> constraintSets) {
072      return constraintSets.isEmpty() ? OscalBindingContext.instance() : new OscalBindingContext(constraintSets);
073    }
074
075    @Override
076    @NonNull
077    public List<Source> getXmlSchemas() throws IOException {
078      return getOscalXmlSchemas();
079    }
080
081    @Override
082    @NonNull
083    public JSONObject getJsonSchema() {
084      return getOscalJsonSchema();
085    }
086  }
087}
