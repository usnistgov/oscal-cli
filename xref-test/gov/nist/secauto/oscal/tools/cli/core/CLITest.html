28  
29  import static org.junit.jupiter.api.Assertions.assertAll;
30  import static org.junit.jupiter.api.Assertions.assertEquals;
31  import static org.junit.jupiter.api.Assertions.assertNull;
32  
33  import gov.nist.secauto.metaschema.cli.processor.ExitCode;
34  import gov.nist.secauto.metaschema.cli.processor.ExitStatus;
35  
36  import org.junit.jupiter.api.Test;
37  
38  import edu.umd.cs.findbugs.annotations.NonNull;
39  
40  public class CLITest {
41  
42    @Test
43    void testVersionInfo() {
44      String[] args = { "--version" };
45      evaluateResult(CLI.runCli(args), ExitCode.OK);
46    }
47  
48    void evaluateResult(@NonNull ExitStatus status, @NonNull ExitCode expectedCode) {
49      status.generateMessage(true);
50      assertAll(
51          () -> assertEquals(expectedCode, status.getExitCode(), "exit code mismatch"),
52          () -> assertNull(status.getThrowable(), "expected null Throwable"));
53    }
54  
55    void evaluateResult(@NonNull ExitStatus status, @NonNull ExitCode expectedCode,
56        @NonNull Class<? extends Throwable> thrownClass) {
57      status.generateMessage(true);
58      Throwable thrown = status.getThrowable();
59      assert thrown != null;
60      assertAll(
61          () -> assertEquals(expectedCode, status.getExitCode(), "exit code mismatch"),
62          () -> assertEquals(thrownClass, thrown.getClass(), "expected Throwable mismatch"));
63    }
64  }
