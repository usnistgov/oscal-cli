43  public class NullYamlValuesTest {
44    private static OscalBindingContext bindingContext;
45    private static IBoundLoader loader;
46  
47    @BeforeAll
48    static void initialize() {
49      bindingContext = OscalBindingContext.instance();
50      loader = bindingContext.newBoundLoader();
51    }
52  
53    @SuppressWarnings("null")
54    @Test
55    void testLoadYamlNullVar1() throws IOException, URISyntaxException {
56      // the YAML catalog is currently malformed, this will create a proper one for
57      // this test
58      AssessmentResults data
59          = loader.load(
60              ObjectUtils.requireNonNull(NullYamlValuesTest.class.getResource("/yaml-null/example_ar_nullvar-1.yaml")));
61  
62      bindingContext.newSerializer(Format.XML, AssessmentResults.class).serialize(data, System.out);
63      bindingContext.newSerializer(Format.JSON, AssessmentResults.class).serialize(data, System.out);
64      bindingContext.newSerializer(Format.YAML, AssessmentResults.class).serialize(data, System.out);
65  
66      assertTrue(data.getResults().get(0).getFindings().isEmpty());
67    }
